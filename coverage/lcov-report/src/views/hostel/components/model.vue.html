<script lang="ts">
import { Component, Prop, Vue } from 'vue-property-decorator'
 
import goBack from '@/components/GoBack/index.vue'
import MaterialInput from '@/components/MaterialInput/index.vue'
 
import { regionData, TextToCode } from 'element-china-area-data'
 
import { hostelAdd, hostelEdit, fetchHostelDetail } from '@/api/hostel'
import { getAliOSS } from '@/api/upLoadOSS'
 
import { rename, deepClone, validateImagesError } from '@/utils'
import { rules } from '@/validate/hostel'
 
import configCode from '@/config/configCode'
import configPrompt from '@/config/configPrompt'
const { HTTP_RESPONSE_STATUS_SUCCESS, DOMAIN } = configCode
const { SU_ADD_HOSTEL, SU_EDIT_HOSTEL, SU_FILE_UPLOAD } = configPrompt
 
import { Action, State, namespace } from 'vuex-class'
const counterModule = namespace('cache')
import { ISetCacheViewData } from '@/store/interface'
 
@Component({
  components: {
    goBack,
    MaterialInput
  }
})
export default class HostelModel extends Vue {
  @Prop({ type: Boolean, default: false }) isEdit!: boolean
  @counterModule.State public cacheData!: { [propName: string]: any }
  @counterModule.Action public SetCacheViewData!: (
    data: ISetCacheViewData
  ) => void
  private listLoading!: boolean
  private tabPosition!: string
  private viewType!: string
  private id!: string | number
  private path!: string
  private url!: any[]
  private dialogImageUrl!: string
  private dialogVisiblePicture!: boolean
  private options!: any
  private domain: string = DOMAIN
  private ossResponse: any = {
    key: [],
    policy: '',
    OSSAccessKeyId: '',
    success_action_status: '200',
    callback: '',
    signature: ''
  }
  private form: { [propName: string]: any } = {
    name: '',
    phone: '',
    region: [],
    province: '',
    city: '',
    county: '',
    address: '',
    note: '',
    logo: '',
    images: []
  }
  private rules: any = rules
  private data() {
    return {
      listLoading: false,
      tabPosition: 'top',
      viewType: 'add',
      id: '',
      path: '',
      url: [],
      dialogImageUrl: '',
      dialogVisiblePicture: false, // 图片预览弹窗初始状态
      options: regionData // 三级联动数据
    }
  }
  get routeHash() {
    return this.$route.hash
  }
  get routeName(): any {
    return this.$route.name
  }
  private created(): void {
    this.initialize()
  }
  private handleReset() {
    return this.$router.go(-1)
  }
  private loading(boolean: boolean): void {
    this.listLoading = boolean
  }
  private refs(ref: string, unitCb?: () => void): any {
    unitCb && unitCb()
    return this.$refs[ref]
  }
  private initOSSAuth(response: Ajax.AxiosResponse) {
    this.path = response.data.dir
    this.ossResponse.key = response.data.dir
    this.ossResponse.policy = response.data.policy
    this.ossResponse.OSSAccessKeyId = response.data.accessid
    this.ossResponse.signature = response.data.signature
    return this.ossResponse
  }
  private handleExceed(files: any, fileList: any) {
    this.$message.warning(
      `当前最多上传 5 张图片，本次选择了 ${
        files.length
      } 张图片，共选择了 ${files.length + fileList.length} 张图片`
    )
  }
  private beforeLogoUpload(file: any): boolean {
    if (validateImagesError(file)) {
      return false
    }
    console.log(file)
    this.ossResponse.key = this.path + rename(file)
    return true
  }
  private beforeImagesUpload(file: any): boolean {
    if (validateImagesError(file)) {
      return false
    }
    this.ossResponse.key = this.path + rename(file)
    this.url.push(this.ossResponse.key)
    return true
  }
  private handleLogoSuccess(res: any, file: any): void {
    this.form.logo = this.ossResponse.key
    this.$message({
      message: SU_FILE_UPLOAD,
      type: 'success'
    })
    this.refs('logoForm').clearValidate()
  }
  private handleImagesSuccess(res: any, file: any, fileList: any[]): void {
    const itemLen = fileList.length
    const arr = []
    for (let i = 0; i < itemLen; i++) {
      arr.push({
        name: fileList[i].name,
        url: this.domain + this.url[i],
        realUrl: this.url[i]
      })
    }
    this.form.images = arr
    this.$message({
      message: SU_FILE_UPLOAD,
      type: 'success'
    })
    this.refs('imagesForm').clearValidate()
  }
  private handleRemove(file: any, fileList: any): void {
    const index = this.form.images.indexOf(file)
    if (index > -1) {
      this.form.images.splice(index, 1)
      this.url.splice(index, 1)
    }
  }
  private handlePictureCardPreview(file: any): void {
    this.dialogImageUrl = file.url
    this.dialogVisiblePicture = true
  }
  // 三级联动改变
  private handleChange(value: any[]): void {
    this.form.province = value[0]
    this.form.city = value[1]
    this.form.county = value[2]
  }
  private goBack() {
    window.history.length > 1 ? this.$router.go(-1) : this.$router.push('/')
  }
  private async initialize() {
    this.setViewType()
    this.initOSSAuth(await getAliOSS())
    if (this.isCacheViewAndhasCacheData()) {
      this.form = this.getProp(this.cacheData, 'from')
    } else {
      this.isEdit && this.getInfo(this.getAndSetId())
    }
  }
  private cacheViewData(): void {
    if (this.routeName) {
      const params = {
        viewName: this.routeName,
        viewType: this.viewType,
        cacheData: {
          from: this.form
        }
      }
      this.SetCacheViewData(params)
    }
  }
  private getProp(tergetData: { [propNanme: string]: any }, prop: string) {
    const val = tergetData[this.routeName][this.viewType][prop]
    if (!val) {
      return false
    }
    return val
  }
  private isCacheViewAndhasCacheData() {
    return (
      this.routeName &&
      this.routeHash &&
      this.cacheData[this.routeName] &&
      this.cacheData[this.routeName][this.viewType]
    )
  }
  private setViewType() {
    this.viewType = this.isEdit ? 'edit' : 'add'
  }
  private getAndSetId(): boolean | string {
    this.id = this.$route.params.id
    return this.id
  }
  private getInfo(id: boolean | string): void {
    this.loading(true)
    const query = {
      id: id
    }
    fetchHostelDetail(query)
      .then((response: Ajax.AxiosResponse) => {
        const status = parseInt((response.data.status) as string, 10)
        if (status === HTTP_RESPONSE_STATUS_SUCCESS) {
          this.assignData(response.data.data)
        } else {
          this.$message({
            message: response.data.data.detail,
            type: 'error'
          })
        }
      })
      .finally(() => {
        this.loading(false)
      })
  }
  private assignData(data: any) {
    const { province, city, county } = data
    const region = []
    region.push(
      TextToCode[province].code,
      TextToCode[province][city].code,
      TextToCode[province][city][county].code
    )
    this.form.id = data.id
    this.form.name = data.name
    this.form.logo = data.logo
    this.form.phone = data.phone
    this.form.province = TextToCode[province].code
    this.form.city = TextToCode[province][city].code
    this.form.county = TextToCode[province][city][county].code
    this.form.region = region
    this.form.address = data.address
    this.form.note = data.note
    const arr = []
    for (const item of data.images) {
      arr.push({
        name: item,
        url: this.domain + item,
        realUrl: item
      })
    }
    this.form.images = arr
    this.url = data.img
  }
  private formSubmit() {
    this.refs('form').validate(this.validateCb)
  }
  private validateCb(vaild: boolean): void {
    if (vaild) {
      const realData = deepClone(this.form)
      realData.images = realData.images.map((item: any) => item.realUrl)
      delete realData.region
      if (this.isEdit) {
        this.handleEdit(realData)
      } else {
        this.handleAdd(realData)
      }
    }
  }
  private handleAdd(data: { [propName: string]: any }): void {
    this.loading(true)
    hostelAdd(data)
      .then((response: Ajax.AxiosResponse) => {
        const status = parseInt((response.data.status) as string, 10)
        if (status === HTTP_RESPONSE_STATUS_SUCCESS) {
          this.$message({
            message: SU_ADD_HOSTEL,
            type: 'success'
          })
          this.goBack()
        } else {
          this.$message({
            message: response.data.data.detail,
            type: 'error'
          })
        }
      })
      .finally(() => {
        this.loading(false)
      })
  }
  private handleEdit(data: { [propName: string]: any }): void {
    this.loading(true)
    hostelEdit(data)
      .then((response: Ajax.AxiosResponse) => {
        const status = parseInt((response.data.status) as string, 10)
        if (status === HTTP_RESPONSE_STATUS_SUCCESS) {
          this.$message({
            message: SU_EDIT_HOSTEL,
            type: 'success'
          })
          this.goBack()
        } else {
          this.$message({
            message: response.data.data.detail,
            type: 'error'
          })
        }
      })
      .finally(() => {
        this.loading(false)
      })
  }
}
</script>
