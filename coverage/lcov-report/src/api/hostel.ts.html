import request from '@/utils/request'
 
export function fetchHostel(query: any) {
  return request({
    url: '/client/hostel/list',
    method: 'GET',
    params: query
  })
}
 
export function fetchHostelDetail(query: any) {
  return request({
    url: '/client/hostel/show',
    method: 'GET',
    params: query
  })
}
 
export function hostelAdd(data: any) {
  return request({
    // url: '/client/hostel/add',
    url: '/client/add',
    method: 'POST',
    data
  })
}
 
export function hostelEdit(data: any) {
  return request({
    // url: '/client/hostel/edit',
