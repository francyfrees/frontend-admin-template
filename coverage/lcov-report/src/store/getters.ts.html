interface IState {
  app: {
    sidebar: {
      opened: boolean
      withoutAnimation: boolean
    }
    sidebarStatus: boolean
    device: string
    isLoading: boolean
    language: string
    theme: string
  }
  tagsView: {
    visitedViews: any[]
    cachedViews: any[]
  }
  user: {
    firstTime: string | number
    auditContent: string
    companyId: string | number
    adminType: string | number
    adminInfo: {
      adminId: string | number
      auditState: number
      newGuidanceState: number
    }
    adminName: string
    adminAvatar: string
    rolesId: string | number
  }
  permission: {
    hasCachePermission: boolean
    permission: {
      modelPermission: number[]
      modelOperationPermission: {
        [propName: string]: string[]
      }
    }
    routers: any[]
  }
  errorLog: {
    logs: any[]
  }
  system: {
    mark: string
  }
}
 
const getters = {
  sidebar: (state: any) => state.app.sidebar,
  language: (state: any) => state.app.language,
  isLoading: (state: any) => state.app.isLoading,
  theme: (state: any) => state.app.theme,
  device: (state: any) => state.app.device,
  sidebarStatus: (state: any) => state.app.sidebarStatus,
 
  visitedViews: (state: any) => state.tagsView.visitedViews,
  cachedViews: (state: any) => state.tagsView.cachedViews,
 
  firstTime: (state: any) => state.user.firstTime,
  auditContent: (state: any) => state.user.auditContent,
  adminInfo: (state: any) => state.user.adminInfo,
  adminAvatar: (state: any) => state.user.adminAvatar,
  adminName: (state: any) => state.user.adminName,
  rolesId: (state: any) => state.user.rolesId,
  companyId: (state: any) => state.user.companyId,
  adminType: (state: any) => state.user.adminType,
 
  hasCachePermission: (state: any) => state.permission.hasCachePermission,
  permission: (state: any) => state.permission.permission,
  permissionRouters: (state: any) => state.permission.routers,
 
  errorLogs: (state: any) => state.errorLog.logs,
 
  mark: (state: any) => state.system.mark
}
export default getters
